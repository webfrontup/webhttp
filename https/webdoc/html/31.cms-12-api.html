1.2 app/controller/base.js #

app/controller/base.js

const Controller = require('egg').Controller;

class BaseController extends Controller {
  async success(data) {
    this.ctx.body={
      code: 0,
      data
    }
  }
  async error(error) {
    this.ctx.body={
      code: 1,
      error
    }
  }
  // 分页 关键字查询 排序
  async index() {
    const { app, ctx, service } = this;
    const {pageNum,pageSize,...where}=ctx.query;
    const list=await service[this.model].list(pageNum,pageSize,where);
    const total=await service[this.model].count(where);
    this.success({
      list,
      total
    });
  }
  async create() {
    const { app, ctx, service } = this;
    const body = ctx.request.body;
    const result = await service[this.model].create(body);
    this.success(result);
  }
  async update() {
    const { app, ctx, service } = this;
    const id = ctx.params.id;
    const body = ctx.request.body;
    body.id = id;
    const result = await service[this.model].update(body);
    this.success(result);
  }
  async destroy() {
    const { app, ctx, service } = this;
    const id=ctx.params.id;
    let ids=ctx.request.body;
    if (!ids) {
      ids=[id];
    }
    const result = await service[this.model].destroy(ids);
    this.success(result);
  }
}

module.exports = BaseController;

1.8 app/service/base.js #

app/service/base.js

const Service = require('egg').Service;

class BaseService extends Service {
    async list(pageNum, pageSize, where) {
        const {app}=this;
        pageNum=isNaN(pageNum)? 1:parseInt(pageNum);
        pageSize=isNaN(pageSize)? 3:parseInt(pageSize);
        const result = await app.mysql.select(this.model, {
            where,
            orders: [['id', 'desc']],
            offset: (pageNum - 1) * pageSize,
            limit: pageSize,
        });
        return result;
    }
    async count(where) {
        return await this.app.mysql.count(this.model,where);
    }
    async create(user) {
        const {app}=this;
        const result = await app.mysql.insert(this.model, user);
        const affectedRows = result.affectedRows;
        return affectedRows > 0 ? result.insertId : 'error';
    }
    async update(updateUser) {
        const { app } = this;
        const result = await app.mysql.update(this.model, updateUser);
        const affectedRows = result.affectedRows;
        return affectedRows > 0 ? 'success' : 'error';
    }
    async destroy(ids) {
        const { app } = this;
        const result = await app.mysql.delete(this.model, { id :ids});
        const affectedRows = result.affectedRows;
        return affectedRows > 0 ? 'success' : 'error';
    }
}

module.exports = BaseService;

    pageSize=isNaN(pageSize)? 3:parseInt(pageSize);
    const roles = await app.mysql.select(this.model, {
        where,

        offset: (pageNum - 1) * pageSize,
        limit: pageSize,
    });
    for (let i=0;i<roles.length;i++){
      let rows=await this.app.mysql.query(`select resource_id from role_resource where role_id=?`,[roles[i].id]);
      roles[i].resourceIds=rows.map(item => item.resource_id);
      //rows=await this.app.mysql.query(`select user_id from role_user where role_id=?`,[roles[i].id]);
      //roles[i].users=rows.map(item=>item.user_id);
    }
    return roles;
}
  async getUsers() {
    return await this.app.mysql.select('user');
  }
  async getResources() {
    let resources = await this.app.mysql.select('resource');
    let map={};
    let top=[];
    resources=resources.filter(resource => {
      resource.children=[];
      map[resource.id]=resource;
      if (resource.parent_id == 0) {
        top.push(resource);
      } else {
        return true;
      }
    });
    resources.forEach(resource => {
      let parentId=resource.parent_id;
      map[parentId].children.push(resource);
    });
    return top;
  }
  async setRolePermission({roleId,resourceIds}) {
    await this.app.mysql.query(`delete from role_resource where role_id =?`,[roleId]);
    for (let i=0;i<resourceIds.length;i++){
      await this.app.mysql.query(`insert into role_resource(role_id,resource_id) values(?,?)`,[roleId,parseInt(resourceIds[i])]);
    }
  }
